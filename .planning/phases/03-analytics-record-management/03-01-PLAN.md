```java
package com.healthy.service;

import com.healthy.entity.AnalyticsData;
import java.time.LocalDate;

public interface AnalyticsService {
    AnalyticsData getIndicatorStats(Long userId, LocalDate startDate, LocalDate endDate);
    AnalyticsData getSportStats(Long userId, LocalDate startDate, LocalDate endDate);
    AnalyticsData getDietStats(Long userId, LocalDate startDate, LocalDate endDate);
    AnalyticsData getSleepStats(Long userId, LocalDate startDate, LocalDate endDate);
}
```

Create `AnalyticsServiceImpl.java` in `impl/` directory:

1. Autowire all 4 mappers: `HealthIndicatorMapper`, `SportRecordMapper`, `DietRecordMapper`, `SleepRecordMapper`

2. For `getIndicatorStats(Long userId, LocalDate startDate, LocalDate endDate)`:
   - Fetch all HealthIndicator records for userId
   - Filter by date range (startDate inclusive, endDate inclusive)
   - Parse bloodPressure string "systolic/diastolic" - return systolic values for the chart
   - Group by recordDate using `Collectors.groupingBy`
   - For each date, take the latest record's values (in case of multiple per day)
   - Compute stats using `DoubleSummaryStatistics` on heartRate and bloodSugar fields
   - Return AnalyticsData with dates sorted ascending, corresponding values

3. For `getSportStats(Long userId, LocalDate startDate, LocalDate endDate)`:
   - Fetch all SportRecord for userId
   - Filter by date range
   - Group by recordDate, sum durations using `Collectors.groupingBy(..., Collectors.summingInt(SportRecord::getDuration))`
   - Return AnalyticsData with dates, durations as doubles, stats (avg/min/max/total of durations)

4. For `getDietStats(Long userId, LocalDate startDate, LocalDate endDate)`:
   - Fetch all DietRecord for userId
   - Filter by date range
   - Group by recordDate, sum calories using `Collectors.groupingBy(..., Collectors.summingDouble(DietRecord::getCalorie))`
   - Return AnalyticsData

5. For `getSleepStats(Long userId, LocalDate startDate, LocalDate endDate)`:
   - Fetch all SleepRecord for userId
   - Filter by date range
   - Group by recordDate (use .toLocalDate() on sleepTime/wakeTime if needed)
   - Sum durations per day
   - Return AnalyticsData

Create `AnalyticsController.java` with 4 GET endpoints:

```java
package com.healthy.controller;

import com.healthy.common.Result;
import com.healthy.entity.AnalyticsData;
import com.healthy.service.AnalyticsService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.format.annotation.DateTimeFormat;
import org.springframework.web.bind.annotation.*;

import java.time.LocalDate;

@RestController
@RequestMapping("/analytics")
public class AnalyticsController {

    @Autowired
    private AnalyticsService analyticsService;

    @GetMapping("/indicator")
    public Result getIndicatorStats(
            @RequestParam Long userId,
            @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate startDate,
            @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate endDate) {
        // Default: last 7 days if not specified
        if (startDate == null) startDate = LocalDate.now().minusDays(6);
        if (endDate == null) endDate = LocalDate.now();
        AnalyticsData data = analyticsService.getIndicatorStats(userId, startDate, endDate);
        return Result.success(data);
    }

    @GetMapping("/sport")
    public Result getSportStats(
            @RequestParam Long userId,
            @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate startDate,
            @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate endDate) {
        if (startDate == null) startDate = LocalDate.now().minusDays(6);
        if (endDate == null) endDate = LocalDate.now();
        AnalyticsData data = analyticsService.getSportStats(userId, startDate, endDate);
        return Result.success(data);
    }

    @GetMapping("/diet")
    public Result getDietStats(
            @RequestParam Long userId,
            @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate startDate,
            @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate endDate) {
        if (startDate == null) startDate = LocalDate.now().minusDays(6);
        if (endDate == null) endDate = LocalDate.now();
        AnalyticsData data = analyticsService.getDietStats(userId, startDate, endDate);
        return Result.success(data);
    }

    @GetMapping("/sleep")
    public Result getSleepStats(
            @RequestParam Long userId,
            @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate startDate,
            @RequestParam(required = false) @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate endDate) {
        if (startDate == null) startDate = LocalDate.now().minusDays(6);
        if (endDate == null) endDate = LocalDate.now();
        AnalyticsData data = analyticsService.getSleepStats(userId, startDate, endDate);
        return Result.success(data);
    }
}
```

Use `@DateTimeFormat(iso = DateTimeFormat.ISO.DATE)` for parsing "YYYY-MM-DD" date strings from frontend. Default to last 7 days when not specified (matching D-07 from CONTEXT: "Default view: last 7 days on first load").
